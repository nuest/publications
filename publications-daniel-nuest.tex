\newrobustcmd{\combinetwobibfilter}[3]{%
 \csedef{temp@bibfilter}{%
  \csuse{save@bibfilter#2} 
  or
  \csuse{save@bibfilter#3} 
  }%
  \defbibfilter{#1}{\csuse{temp@bibfilter}}%
}
